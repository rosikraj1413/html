import React, { useState } from "react";
import { useNavigate } from "react-router-dom";

const App = () => {
  const navigate = useNavigate();
  const [showLoginOptions, setShowLoginOptions] = useState(false);

  const handleLoginClick = () => {
    setShowLoginOptions(!showLoginOptions);
  };

  const handleLogin = (userType) => {
    if (userType === "customer") {
      navigate('/customer-login');
    } else if (userType === "vendor") {
      navigate('/vendor-login');
    }
    setShowLoginOptions(false);
  };

  const handleRegister = (userType) => {
    if (userType === "customer") {
      navigate('/customer-register');
    } else if (userType === "vendor") {
      navigate('/vendor-register');
    }
  };

  return (
    <div>
      <nav className="navbar navbar-expand-lg navbar-light bg-light">
        <div className="container-fluid">
          <a className="navbar-brand" href="#">
            Pharma Easy
          </a>
          <button
            className="navbar-toggler"
            type="button"
            data-bs-toggle="collapse"
            data-bs-target="#navbarNav"
            aria-controls="navbarNav"
            aria-expanded="false"
            aria-label="Toggle navigation"
          >
            <span className="navbar-toggler-icon"></span>
          </button>
          <div className="collapse navbar-collapse" id="navbarNav">
            <ul className="navbar-nav ml-auto">
              <li className="nav-item dropdown">
                <a
                  className="nav-link dropdown-toggle"
                  href="#"
                  id="navbarDropdown"
                  role="button"
                  data-bs-toggle="dropdown"
                  aria-expanded="false"
                  onClick={handleLoginClick}
                >
                  Login
                </a>
                <ul className={`dropdown-menu ${showLoginOptions ? "show" : ""}`} aria-labelledby="navbarDropdown">
                  <li>
                    <a className="dropdown-item" onClick={() => handleLogin("customer")}>
                      Customer
                    </a>
                  </li>
                  <li>
                    <a className="dropdown-item" onClick={() => handleLogin("vendor")}>
                      Vendor
                    </a>
                  </li>
                </ul>
              </li>
              <li className="nav-item">
                <a className="nav-link" onClick={() => handleRegister("customer")}>
                  Register Customer
                </a>
              </li>
              <li className="nav-item">
                <a className="nav-link" onClick={() => handleRegister("vendor")}>
                  Register Vendor
                </a>
              </li>
            </ul>
          </div>
        </div>
      </nav>

      <div className="container">
        {/* Your other content */}
        <h1 className="mt-4">Pharma Easy</h1>
        <p>Get your Medicines to your door step Anywhere AnyTime</p>
      </div>
    </div>
  );
};

export default App;
